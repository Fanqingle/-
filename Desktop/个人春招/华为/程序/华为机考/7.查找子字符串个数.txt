int fun(char *str, char *substr)
{
	int count = 0;
	
	while (*str)
	{
		char *p = str;
		char *q = substr;
		
		while (*q)
		{
			if (*p == *q)
			{
				p++;
				q++;
			}
			else
			{
				break;
			}
		}
		
		if (*q == 0)
		{
			str = str + strlen(substr);
			count++;
		}
		else
		{
			str++;
		}
	}
	return count;
}
